TÂCHE 3 : MODIFIER ROUTE GET /api/invoices
Endpoint : GET /api/invoices
Modification : Joindre la table categories pour récupérer les infos de catégorie.
Requête SQL (avec Drizzle ORM) :
const userInvoices = await db
  .select({
    // Tous les champs invoices
    id: invoices.id,
    user_id: invoices.user_id,
    supplier_id: invoices.supplier_id,
    amount_display_ttc: invoices.amount_display_ttc,
    amount_ht: invoices.amount_ht,
    amount_real_ttc: invoices.amount_real_ttc,
    is_stock_purchase: invoices.is_stock_purchase,
    category_id: invoices.category_id,
    vat_applicable: invoices.vat_applicable,
    has_brs: invoices.has_brs,
    invoice_type: invoices.invoice_type,
    invoice_number: invoices.invoice_number,
    description: invoices.description,
    date: invoices.date,
    project_id: invoices.project_id,
    payment_method: invoices.payment_method,
    file_url: invoices.file_url,
    created_at: invoices.created_at,
    
    // Champs pour rétrocompatibilité
    category: invoices.category, // ancien champ texte
    amount: invoices.amount, // ancien champ montant
    
    // Jointure categories
    category_app_name: categories.app_name,
    category_account_name: categories.account_name,
    category_account_code: categories.account_code,
    
    // Jointure suppliers
    supplier_name: suppliers.name,
    supplier_is_regular: suppliers.is_regular_supplier
  })
  .from(invoices)
  .leftJoin(categories, eq(invoices.category_id, categories.id))
  .leftJoin(suppliers, eq(invoices.supplier_id, suppliers.id))
  .where(eq(invoices.user_id, userId))
  .orderBy(desc(invoices.date));
