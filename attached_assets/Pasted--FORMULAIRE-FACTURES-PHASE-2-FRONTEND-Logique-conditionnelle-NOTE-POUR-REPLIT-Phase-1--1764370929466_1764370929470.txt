useEffect(() => {

    setAmountHT(null);
    setShowAmountHT(false);
  }

useEffect 4 : Affichage champ BRS
useEffect(() => {
  if (!selectedCategory) return;
  
  const isPrestationServices = selectedCategory.account_name === "Achats d'études et prestations de services";
  
  if (!vatApplicable && isPrestationServices) {
    setShowBRSField(true);
  } else {
    setShowBRSField(false);
    setHasBRS(false); // reset
    setShowRealAmountField(false); // reset
  }
}, [vatApplicable, selectedCategory]);
useEffect 5 : Calcul Montant TTC réel (BRS)
useEffect(() => {
  if (hasBRS && showBRSField && amountDisplayTTC > 0) {
    const realTTC = parseFloat(amountDisplayTTC) / 0.95;
    setAmountRealTTC(realTTC);
    setShowRealAmountField(false); // ne pas afficher si BRS = Oui
  } else if (showBRSField && !hasBRS && amountDisplayTTC > 0) {
    // BRS = Non, afficher le champ montant réel
    const realTTC = parseFloat(amountDisplayTTC) / 0.95;
    setAmountRealTTC(realTTC);
    setShowRealAmountField(true);
  } else {
    setAmountRealTTC(null);
    setShowRealAmountField(false);
  }
}, [hasBRS, showBRSField, amountDisplayTTC]);
useEffect 6 : Forcing Type de facture (logique complexe)
useEffect(() => {
  const amount = parseFloat(amountDisplayTTC) || 0;

const handleSubmit = async (e) => {
  e.preventDefault();
  
  // Validation frontend basique
  if (!supplierId || !amountDisplayTTC || !categoryId || !invoiceType) {
    alert('Veuillez remplir tous les champs obligatoires');
    return;
  }
  
  if (invoiceType === 'supplier_invoice' && !invoiceNumber.trim()) {
    alert('Numéro de facture obligatoire pour les Factures Fournisseur');
    return;
  }
  
  // Préparer les données
  const invoiceData = {
    supplier_id: parseInt(supplierId),
    amount_display_ttc: parseFloat(amountDisplayTTC),
    is_stock_purchase: isStockPurchase,
    category_id: parseInt(categoryId),
    vat_applicable: vatApplicable,
    has_brs: hasBRS,
    invoice_type: invoiceType,
    invoice_number: invoiceType === 'supplier_invoice' ? invoiceNumber : null,
    description,
    date,
    project_id: parseInt(projectId),
    payment_method: paymentMethod,
    file_url: uploadedFileUrl // ou file selon upload existant
  };
  
  try {
    const response = await fetch('/api/invoices', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(invoiceData)
    });
    
    if (!response.ok) {
      const error = await response.json();
      alert(`Erreur : ${error.error || 'Erreur lors de la soumission'}`);
      return;
    }
    
    const result = await response.json();
    alert('Facture soumise avec succès !');
    // Rediriger ou réinitialiser le formulaire
    
  } catch (error) {
    console.error('Erreur soumission:', error);
    alert('Erreur lors de la soumission de la facture');
  }
};
