## 1. BACKEND - Route GET /api/invoices (server/routes.ts)

### Problème actuel
La route renvoie les factures SANS les champs calculés `totalPaid` et `remainingAmount`.

### Solution requise

**Modifier la requête SQL pour JOIN avec payments et calculer les totaux :**

```typescript
// Dans server/routes.ts - Route GET /api/invoices

app.get("/api/invoices", async (req, res) => {
  const { userName } = req.query;
  
  try {
    // Récupérer les factures avec calcul des paiements
    const invoicesQuery = `
      SELECT 
        i.*,
        COALESCE(SUM(CAST(p.amount_paid AS NUMERIC)), 0) as total_paid,
        CASE 
          WHEN i.invoice_type = 'supplier_invoice' 
          THEN CAST(COALESCE(i.amount_display_ttc, i.amount_ttc) AS NUMERIC) - COALESCE(SUM(CAST(p.amount_paid AS NUMERIC)), 0)
          ELSE NULL
        END as remaining_amount
      FROM invoices i
      LEFT JOIN payments p ON i.id = p.invoice_id
      WHERE i.user_name = $1
      GROUP BY i.id
      ORDER BY i.invoice_date DESC
    `;
    
    const result = await db.query(invoicesQuery, [userName]);
    
    // Formater les résultats
    const invoices = result.rows.map(invoice => ({
      ...invoice,
      totalPaid: invoice.total_paid ? parseFloat(invoice.total_paid) : 0,
      remainingAmount: invoice.remaining_amount ? parseFloat(invoice.remaining_amount) : null
    }));
    
    res.json(invoices);
  } catch (error) {
    console.error("Error fetching invoices:", error);
    res.status(500).json({ error: "Failed to fetch invoices" });
  }
});
IMPORTANT :
Utiliser COALESCE pour gérer les factures sans paiements (retourne 0)
Utiliser CAST AS NUMERIC pour les calculs (les montants sont stockés en VARCHAR)
Calculer remaining_amount UNIQUEMENT pour les factures fournisseur
Le LEFT JOIN permet d'inclure les factures sans paiements

5. VÉRIFICATION - Modal AddPaymentModal existe-t-elle ?
Si la modal AddPaymentModal n'existe PAS encore :
Créer client/src/components/AddPaymentModal.tsx :
import { useState } from "react";
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { useToast } from "@/hooks/use-toast";

interface AddPaymentModalProps {
  invoice: any;
  isOpen: boolean;
  onClose: () => void;
  onSuccess: (invoice: any) => void;
}

export function AddPaymentModal({ invoice, isOpen, onClose, onSuccess }: AddPaymentModalProps) {
  const [amountPaid, setAmountPaid] = useState("");
  const [paymentDate, setPaymentDate] = useState(new Date().toISOString().split('T')[0]);
  const [paymentType, setPaymentType] = useState("");
  const [isSubmitting, setIsSubmitting] = useState(false);
  const { toast } = useToast();

  const totalPaid = invoice.totalPaid || 0;
  const amountToPay = parseFloat(invoice.amountDisplayTTC || invoice.amountTTC);
  const remaining = amountToPay - totalPaid;

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    const amount = parseFloat(amountPaid);
    
    // Validations
    if (isNaN(amount) || amount <= 0) {
      toast({ title: "Erreur", description: "Le montant doit être supérieur à 0", variant: "destructive" });
      return;
    }
    
    if (amount > remaining) {
      toast({ title: "Erreur", description: `Le montant ne peut pas dépasser ${remaining.toLocaleString('fr-FR')} F`, variant: "destructive" });
      return;
    }
    
    if (!paymentDate || !paymentType) {
      toast({ title: "Erreur", description: "Tous les champs sont requis", variant: "destructive" });
      return;
    }

    setIsSubmitting(true);
    
    try {
      const response = await fetch("/api/payments", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          invoiceId: invoice.id,
          amountPaid: amountPaid,
          paymentDate: paymentDate,
          paymentType: paymentType
        })
      });

      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.message || "Erreur lors de l'enregistrement");
      }

      const result = await response.json();
      onSuccess(result.invoice);
    } catch (error: any) {
      toast({ title: "Erreur", description: error.message, variant: "destructive" });
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <Dialog open={isOpen} onOpenChange={onClose}>
      <DialogContent className="sm:max-w-md">
        <DialogHeader>
          <DialogTitle>Ajouter un paiement</DialogTitle>
          <p className="text-sm text-muted-foreground">
            Facture {invoice.invoiceNumber} - {invoice.supplierName}
          </p>
        </DialogHeader>

        <div className="space-y-3 py-4">
          <div className="space-y-1">
            <p className="text-sm">Montant TTC : <span className="font-medium">{amountToPay.toLocaleString('fr-FR')} F</span></p>
            <p className="text-sm">Total déjà payé : <span className="font-medium">{totalPaid.toLocaleString('fr-FR')} F</span></p>
            <p className="text-sm font-bold text-teal-600">
              Reste à payer : {remaining.toLocaleString('fr-FR')} F
            </p>
          </div>

          <form onSubmit={handleSubmit} className="space-y-4">
            <div>
              <Label htmlFor="amount">Montant de ce paiement (FCFA) *</Label>
              <Input
                id="amount"
                type="number"
                value={amountPaid}
                onChange={(e) => setAmountPaid(e.target.value)}
                placeholder="Ex: 30000"
                max={remaining}
                required
              />
            </div>

            <div>
              <Label htmlFor="date">Date du paiement *</Label>
              <Input
                id="date"
                type="date"
                value={paymentDate}
                onChange={(e) => setPaymentDate(e.target.value)}
                required
              />
            </div>

            <div>
              <Label htmlFor="type">Type de règlement *</Label>
              <Select value={paymentType} onValueChange={setPaymentType} required>
                <SelectTrigger>
                  <SelectValue placeholder="Sélectionner..." />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value="Wave Business">Wave Business</SelectItem>
                  <SelectItem value="Espèces">Espèces</SelectItem>
                  <SelectItem value="Virement">Virement</SelectItem>
                  <SelectItem value="Mobile Money">Mobile Money</SelectItem>
                </SelectContent>
              </Select>
            </div>

            <DialogFooter>
              <Button type="button" variant="outline" onClick={onClose}>
                Annuler
              </Button>
              <Button type="submit" disabled={isSubmitting}>
                {isSubmitting ? "Enregistrement..." : "Enregistrer"}
              </Button>
            </DialogFooter>
          </form>
        </div>
      </DialogContent>
    </Dialog>
  );
}
